\begin{enumerate}

  by year of creation
\item

  not
\item

  their content

  \begin{itemize}
  \tightlist
  \item
    \texttt{0}: 0 \textless{}= length \textless{} 40~(short)
  \item
    \texttt{1}: 40 \textless{}= length \textless{} 80~(one liner)
  \item
    \texttt{2}: 80 \textless{}= length \textless{} 160~(tweet)
  \item
    \texttt{3}: 160 \textless{}= length~(long)
  \end{itemize}
\end{enumerate}
